15def load_tag_lookup(filename: str) -> Dict[Tuple[int, str], str]: 

16 """ 

17 Loads the tag lookup information from a CSV file. 

18 

19 The CSV file is expected to have three columns: 

20 'dstport', 'protocol', and 'tag'. 

21 The function reads the CSV file, skips the header, 

22 and maps each (dstport, protocol) pair to the corresponding tag. 

23 

24 Args: 

25 filename (str): 

26 The path to the CSV file containing the tag lookup data. 

27 

28 Returns: 

29 Dict[Tuple[int, str], str]: 

30 A dictionary mapping (dstport, protocol) tuples to tags. 

31 

32 Raises: 

33 FileNotFoundError: If the specified file does not exist. 

34 Exception: If there is an error reading the file or 

35 processing its contents. 

36 """ 

37 lookup = {} 

38 try: 

39 with open(filename, mode='r') as file: 

40 reader = csv.reader(file) 

41 next(reader) # Skip header 

42 for row in reader: 

43 dstport, protocol, tag = row 

44 lookup[(int(dstport), protocol.lower())] = tag 

45 logging.info(f"Loaded {len(lookup)} tag mappings from {filename}") 

46 except FileNotFoundError: 

47 logging.error(f"Lookup file not found: {filename}") 

48 sys.exit(1) 

49 except Exception as e: 

50 logging.error(f"Error loading lookup file: {e}") 

51 sys.exit(1) 

52 return lookup

76 """ 

77 with open(filename, 'r') as f:

81def parse_flow_logs( 

82 filename: str, 

83 lookup: Dict[Tuple[int, str], str], 

84 protocol_map: dict) -> Tuple[Dict[str, int], 

85 Dict[Tuple[int, str], int]]: 

86 """ 

87 Parses the flow logs from the given file and generates counts for tags and 

88 port/protocol combinations. 

89 

90 The function reads the flow log file line by line, extracts 

91 the destination port and protocol, and uses the provided lookup 

92 and protocol map to categorize and count the occurrences of each tag 

93 and port/protocol combination. 

94 

95 Args: 

96 filename (str): The path to the flow log file. 

97 lookup (Dict[Tuple[int, str], str]): 

98 A dictionary mapping (dstport, protocol) tuples to tags. 

99 

100 protocol_map (dict): 

101 A dictionary mapping protocol identifiers to protocol names. 

102 

103 Returns: 

104 Tuple[Dict[str, int], Dict[Tuple[int, str], int]]: 

105 - A dictionary mapping tags to their counts. 

106 - A dictionary mapping (dstport, protocol) tuples to their counts. 

107 

108 Raises: 

109 FileNotFoundError: If the specified file does not exist. 

110 Exception: If there is an error reading the file or 

111 processing its contents. 

112 """ 

113 tag_counts: DefaultDict[str, int] = defaultdict(int) 

114 combination_counts: DefaultDict[Tuple[int, str], int] = defaultdict(int) 

115 try: 

116 with open(filename, mode='r') as file: 

117 for line in file: 

118 parts = line.strip().split() 

119 if len(parts) < 8: 

120 logging.warning(f"Skipping invalid line: {line.strip()}") 

121 continue # Skip invalid lines 

122 dstport = int(parts[6]) 

123 protocol = protocol_map.get(parts[7], 'unknown') 

124 combination_counts[(dstport, protocol)] += 1 

125 tag = lookup.get((dstport, protocol), "Untagged") 

126 tag_counts[tag] += 1 

127 logging.info(f"Parsed flow logs from {filename}") 

128 except FileNotFoundError: 

129 logging.error(f"Flow log file not found: {filename}") 

130 sys.exit(1) 

131 except Exception as e: 

132 logging.error(f"Error parsing flow log file: {e}") 

133 sys.exit(1) 

134 return tag_counts, combination_counts

137def generate_output(tag_counts: Dict[str, int], 

138 combination_counts: Dict[Tuple[int, str], int]) -> None: 

139 """ 

140 Generates an output file containing counts of tags and 

141 port/protocol combinations. 

142 

143 This function writes the provided tag counts and 

144 port/protocol combination counts to an output file named `output.txt`. 

145 The output file contains two sections: 

146 1. **Tag Counts**: 

147 Lists each tag and its corresponding count. 

148 2. **Port/Protocol Combination Counts**: 

149 Lists each port/protocol combination and its corresponding count. 

150 

151 Args: 

152 tag_counts (Dict[str, int]): 

153 A dictionary where keys are tags and values 

154 are the counts of each tag. 

155 combination_counts (Dict[Tuple[int, str], int]): 

156 A dictionary where keys are tuples of (port, protocol) and values 

157 are the counts of each port/protocol combination. 

158 

159 Raises: 

160 Exception: If there is an error writing to the output file. 

161 """ 

162 try: 

163 with open('output.txt', 'w') as file: 

164 file.write("Tag Counts:\n\n") 

165 file.write("Tag\t\tCount\n") 

166 for tag, count in tag_counts.items(): 

167 file.write(f"{tag}\t\t{count}\n") 

168 

169 file.write("\nPort/Protocol Combination Counts:\n\n") 

170 file.write("Port\tProtocol\tCount\n") 

171 for (port, protocol), count in combination_counts.items(): 

172 file.write(f"{port}\t{protocol}\t{count}\n") 

173 logging.info("Generated output.txt with tag and combination counts") 

174 except Exception as e: 

175 logging.error(f"Error generating output file: {e}") 

176 sys.exit(1)
